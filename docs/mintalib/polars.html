 15def wrap_polars(result, name: str):
 16    """ wrap result to polars """
 17
 18    if isinstance(result, tuple) and hasattr(result, '_asdict'):
 19        result = result._asdict()
 20
 21    if isinstance(result, dict):
 22        return pl.DataFrame(result).to_struct(name=name)
 23
 24    if isinstance(result, np.ndarray):
 25        return pl.Series(values=result, name=name)

 31def expression_method(func):
 32    name = func.__name__
 33    name = name.removeprefix("calc_")
 34
 35    metadata = getattr(func, 'metadata', {})
 36    output_names = metadata.get('output_names', ())
 37    output_type = Struct({n: Float64 for n in output_names}) if output_names else Float64
 38 
 39    signature = inspect.signature(func)
 40    fparam, *params = signature.parameters.values()
 41    force_struct = fparam.name == 'prices'
 42    newsig = inspect.Signature(params)
 43
 44    def wrapper(self, *args, **kwargs):
 45        expr = self._expr
 46
 47        bound_args = newsig.bind(*args, **kwargs)
 48        args, kwargs = (), dict(bound_args.arguments)
 49
 50        if force_struct and expr.meta.has_multiple_outputs():
 51            expr = pl.struct(expr)
 52
 53        def batch_func(data):
 54            if force_struct:
 55                data = data.struct.unnest()
 56
 57            output = func(data, *args, **kwargs)
 58            return wrap_polars(output, name=name)
 59
 60        result = expr.map_batches(batch_func, return_dtype=output_type)
 61
 62        return result
 63
 64    return wrapper

 98def setup():
 99    from . import core
100    calc_methods = [m for m in dir(core) if m.startswith("calc_")]
101
102    for cname in calc_methods:
103        name = cname.removeprefix("calc_")

106        fparam = list(sig.parameters.values())[0]
107
108        wrapper = expression_method(func)
109        setattr(ExpressionCalc, name, wrapper)
